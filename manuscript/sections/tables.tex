% Table 1: Primary Comparison - The Central Finding
% ============================================================================
\begin{table}[ht]
\centering
\caption{\textbf{Network influence dissociates from target proximity.}
Hyperforin achieves higher DILI-module influence than Quercetin despite fewer targets.
Z-scores computed against degree-matched null distributions ($n = 1{,}000$ permutations; one-tailed; FDR-corrected).
Network: STRING v12.0 LCC (confidence $\geq$900; 7{,}677 nodes, 66{,}908 edges) filtered to liver-expressed genes (GTEx v8; TPM $\geq$1).}
\label{tab:main_results}
\vspace{0.5em}
\small
\begin{tabular}{@{}llccccc@{}}
\toprule
\textbf{Metric} & \textbf{Compound} & \textbf{Targets} & \textbf{Observed} & \textbf{Z} & \textbf{$p$} & \textbf{PTNI} \\
\midrule
\multicolumn{7}{@{}l}{\textit{Tier 1: Shortest-path proximity (descriptive)}} \\[0.3em]
& Hyperforin & 10 & $d_c = 1.30$ & $-$3.86 & $<10^{-16}$$^\dagger$ & --- \\
& Quercetin & 62 & $d_c = 1.68$ & $-$5.44 & $<10^{-16}$$^\dagger$ & --- \\
& \multicolumn{5}{r}{\textit{Proximity ranking is threshold-dependent (reverses at $\geq$700)}} \\[0.5em]
\multicolumn{7}{@{}l}{\textit{Tier 2: Random walk influence (inferential)}} \\[0.3em]
& Hyperforin & 10 & 0.114 & \textbf{+10.27} & $<10^{-16}$$^\dagger$ & 0.0114 \\
& Quercetin & 62 & 0.032 & +4.42 & $4.9 \times 10^{-6}$ & 0.0005 \\
& \multicolumn{5}{r}{\textit{Resolution: Fewer targets $\rightarrow$ stronger influence}} \\[0.5em]
\multicolumn{7}{@{}l}{\textit{Tier 3: Expression-weighted influence (constraining)}} \\[0.3em]
& Hyperforin & 10 & 0.133 & \textbf{+9.07} & $<10^{-16}$$^\dagger$ & 0.0133 \\
& Quercetin & 62 & 0.049 & +5.56 & $1.4 \times 10^{-8}$ & 0.0008 \\
\bottomrule
\end{tabular}

\vspace{0.3em}
\begin{minipage}{\linewidth}
\footnotesize
$^\dagger$At permutation floor ($<$1/1,000); reported as $p < 10^{-16}$ assuming continuous null.\\
PTNI = per-target network influence (Eq.~7); $d_c$ = mean minimum shortest-path distance.
\end{minipage}
\end{table}

% Table 2: Effect Size - The 22x Disparity
% ============================================================================
\begin{table}[ht]
\centering
\caption{\textbf{Per-target influence efficiency.}
PTNI normalizes total DILI-module influence by target count, quantifying efficiency rather than coverage.
Hyperforin targets contribute 17--22$\times$ more influence per target than Quercetin targets.}
\label{tab:ptni}
\vspace{0.5em}
\begin{tabular}{@{}lcccc@{}}
\toprule
\textbf{Analysis} & \textbf{Hyperforin PTNI} & \textbf{Quercetin PTNI} & \textbf{Fold difference} & \textbf{95\% CI}$^*$ \\
\midrule
RWR (topology-only) & 0.0114 & 0.0005 & \textbf{22.8}$\times$ & --- \\
EWI (expression-weighted) & 0.0133 & 0.0008 & \textbf{16.6}$\times$ & --- \\
\bottomrule
\end{tabular}

\vspace{0.3em}
\begin{minipage}{\linewidth}
\footnotesize
$^*$PTNI is a derived normalization, not subjected to independent permutation testing. Advantage persists under expression constraint, indicating biological relevance.
\end{minipage}
\end{table}
